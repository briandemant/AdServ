  var mix = function(defaults, source) {
    var result = {};
    var k;
    for (k in defaults) {
      if (defaults.hasOwnProperty(k)) {
        result[k] = defaults[k];
      }
    }
    for (k in source) {
      if (source.hasOwnProperty(k)) {
        result[k] = source[k];
      }
    }
    return result;
