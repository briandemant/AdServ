Test if an element is visible (searches up the tree until BODY is reached)

var isVisible = AdServ.isVisible = function(elemOrSelector) { var elem = $(elemOrSelector); if (!elem) { return false; }

Body must be visible (anything else would be silly)

if (elem.nodeName === 'BODY') { return true; }

Is it hidden from sight?

if (css(elem, 'visibility') == 'hidden') { return false; }

Is it displayed?

if (css(elem, 'display') == 'none') { return false; }

Is it transparent?

if (css(elem, 'opacity') == '0') { return false; }

Look up the tree

return isVisible(elem.parentNode); };
